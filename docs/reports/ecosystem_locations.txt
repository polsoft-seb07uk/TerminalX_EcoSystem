============================================================
  TerminalX EcoSystem — Lokalizacje folderów i plików
  polsoft.ITS(TM) Group
============================================================

Legenda:
  ROOT_DIR   = folder główny EcoSystem (tam gdzie TerminalX.py)
  HOME       = katalog domowy użytkownika (~)
  __file__   = lokalizacja konkretnego modułu (.py)

------------------------------------------------------------
  STRUKTURA GŁÓWNA (ROOT_DIR)
------------------------------------------------------------

  ROOT_DIR\                         — katalog główny EcoSystem
  ROOT_DIR\TerminalX.py             — punkt wejścia
  ROOT_DIR\core\                    — moduły systemowe
  ROOT_DIR\modules\                 — moduły użytkownika
  ROOT_DIR\plugins\                 — pluginy
  ROOT_DIR\lang\                    — pliki lokalizacji (pl.py, en.py)
  ROOT_DIR\libs\                    — biblioteki zewnętrzne
  ROOT_DIR\libs\pip\                — paczki pip (lokalny cache)
  ROOT_DIR\libs\requirements.txt    — lista wymaganych pakietów
  ROOT_DIR\scripts\                 — skrypty użytkownika
  ROOT_DIR\scripts\logs\            — logi wykonania skryptów (runner.py)
  ROOT_DIR\docs\                    — dokumentacja
  ROOT_DIR\docs\reports\            — raporty systemowe (monitor, analyser)
  ROOT_DIR\gui\                     — aplikacja GUI
  ROOT_DIR\gui\app.py               — główne okno GUI
  ROOT_DIR\tests\                   — testy jednostkowe
  ROOT_DIR\tests\test_core.py       — testy core
  ROOT_DIR\tests\test_i18n.py       — testy lokalizacji
  ROOT_DIR\tests\test_terminalx.py  — testy terminala

  ROOT_DIR\.cache\analyser\                 — cache analizatora
  ROOT_DIR\.cache\analyser\last_report.json — ostatni raport analizy
  ROOT_DIR\.cache\defender\                 — cache defender
  ROOT_DIR\.cache\defender\events.json      — log zdarzeń defender
  ROOT_DIR\.cache\defender\baseline.json    — baseline skrótów SHA-256
  ROOT_DIR\.cache\docs\                     — cache dokumentacji
  ROOT_DIR\.cache\docs\index.json           — indeks docs
  ROOT_DIR\.cache\math\                     — cache obliczeń matematycznych
  ROOT_DIR\.cache\pkg\                      — cache menedżera pakietów
  ROOT_DIR\.cache\pkg\installed.json        — lista zainstalowanych paczek
  ROOT_DIR\.cache\pkg\history.json          — historia instalacji
  ROOT_DIR\.cache\pkg\tmp\                  — pliki tymczasowe pkg
  ROOT_DIR\.cache\scripts\                  — cache skryptów
